interface ShareProps {

  name: { // 即实例id，必填
    type: String
  }
  url: { // 对应地址，必填
    type: String
  }
  props: { // 传给子应用的数据
    type: PropType<any>

  }
  proxy: { // 使用的沙箱
    type: PropType<ProxyGlobals>
  }
  keepAlive: { // keep-alive
    type: Boolean
    default: false
  }
  iframe: { // iframeId
    type: String
  }
}
